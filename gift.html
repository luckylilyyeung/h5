    // 复制工具函数（兼容微信移动端）

    function copyText(text) {

      return new Promise((resolve, reject) => {

        // 现代剪贴板API

        if (navigator.clipboard && window.isSecureContext) {

          navigator.clipboard.writeText(text).then(resolve).catch(reject);

        } else {

          // 兼容低版本微信

          const textarea = document.createElement('textarea');

          textarea.value = text;

          textarea.style.position = 'fixed';

          textarea.style.left = '-9999px';

          document.body.appendChild(textarea);

          textarea.select();

          textarea.setSelectionRange(0, text.length);

          const ok = document.execCommand('copy');

          document.body.removeChild(textarea);

          ok ? resolve() : reject();

        }

      })

    }


    // 弹窗提示

    function showToast(msg) {

      const toast = document.getElementById('toast');

      toast.innerText = msg;

      toast.style.display = 'block';

      setTimeout(() => toast.style.display = 'none', 1500);

    }


    // 打开链接

    function openLink(url) {

      window.location.href = url;

    }


    // 预设按钮绑定

    document.querySelectorAll('.btn-open').forEach(btn => {

      btn.addEventListener('click', () => {

        const link = btn.dataset.link;

        if(link) openLink(link);

      })

    })

    document.querySelectorAll('.btn-copy').forEach(btn => {

      btn.addEventListener('click', async () => {

        const link = btn.dataset.link;

        if(!link) return;

        try {

          await copyText(link);

          showToast('链接复制成功，去浏览器打开');

        } catch (err) {

          showToast('复制失败，请长按链接手动复制');

        }

      })

    })


    // 自定义链接按钮

    const urlInput = document.getElementById('customUrl');

    document.getElementById('customOpen').addEventListener('click', () => {

      const val = urlInput.value.trim();

      if(!val) return showToast('请先输入商品链接');

      openLink(val);

    })

    document.getElementById('customCopy').addEventListener('click', async () => {

      const val = urlInput.value.trim();

      if(!val) return showToast('请先输入商品链接');

      try {

        await copyText(val);

        showToast('自定义链接复制成功');

      } catch (err) {

        showToast('复制失败，长按输入框复制文字');

      }

    })
