function copyText(text){return new Promise((resolve,reject)=>{if(navigator.clipboard){navigator.clipboard.writeText(text).then(resolve).catch(reject);}else{const t=document.createElement('textarea');t.value=text;t.style.left='-9999px';document.body.appendChild(t);t.select();const ok=document.execCommand('copy');document.body.removeChild(t);ok?resolve():reject();}})}

function showToast(msg){const t=document.getElementById('toast');t.innerText=msg;t.style.display='block';setTimeout(()=>t.style.display='none',1500);}
